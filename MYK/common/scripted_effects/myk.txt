update_income_gui = {
	if = {
		limit = {
			check_variable = {
				global.income_gui = 0
			}
		}
		set_variable = { global.income_gui = 1 }
	}
	else_if = {
		limit = {
			check_variable = {
				global.income_gui = 1
			}
		}
		set_variable = { global.income_gui = 0 }
	}
}

	}
